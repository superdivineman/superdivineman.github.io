2. Tensor Basics

Tensors are PyTorch's core data structure — multi-dimensional arrays that can run on GPU.

import torch

# Create tensors
x = torch.tensor([[1.0, 2.0], [3.0, 4.0]])       # from data
zeros = torch.zeros(3, 4)                          # 3x4 zeros
rand = torch.randn(100, 512)                        # random normal
ones = torch.ones(2, 3)                             # 2x3 ones

# Move to GPU (if available)
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
x = x.to(device)

# Operations
y = x.sum(dim=1)              # sum along dim=1
z = x.view(1, -1)             # reshape to 1D
loss = x.mean()              # average
x.retain_grad()              # needed for autograd on non-scalar

# NumPy bridge
np_array = x.cpu().numpy()   # tensor → numpy (must be on CPU)
back_to_tensor = torch.from_numpy(np_array)

3. Autograd & Backpropagation

PyTorch's automatic differentiation engine — it computes gradients automatically.

# Enable gradient tracking
x = torch.tensor([2.0, 3.0], requires_grad=True)

# Forward pass
y = x ** 2 + 3 * x + 1
loss = y.sum()

# Backward pass — computes gradients d(loss)/d(x)
loss.backward()

# Gradient: d(loss)/d(x) = 2x + 3 → [7, 9] when x=[2, 3]
print(x.grad)  # tensor([7., 9.])

# Zero gradients before next backward pass
optimizer.zero_grad()
loss.backward()
optimizer.step()
Key rule: requires_grad=True tells PyTorch to track operations on this tensor. Every tensor that participates in a loss computation and needs gradients should have this set.

4. Neural Network Module (torch.nn)

import torch.nn as nn

class SimpleMLP(nn.Module):
    def __init__(self, input_size, hidden_size, num_classes):
        super().__init__()
        self.layer1 = nn.Linear(input_size, hidden_size)
        self.relu = nn.ReLU()
        self.layer2 = nn.Linear(hidden_size, num_classes)
        self.dropout = nn.Dropout(p=0.3)

    def forward(self, x):
        x = self.layer1(x)
        x = self.relu(x)
        x = self.dropout(x)
        x = self.layer2(x)
        return x

# Instantiate
model = SimpleMLP(input_size=784, hidden_size=256, num_classes=10)
print(model)
print(model.parameters())        # generator of all parameters
print(sum(p.numel() for p in model.parameters()))  # total params

5. Training Loop

# Move model to device
model = model.to(device)

# Loss and optimizer
criterion = nn.CrossEntropyLoss()
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=0.01)
scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=10)

# Training loop
for epoch in range(num_epochs):
    model.train()  # set to training mode (enables dropout, etc.)
    total_loss = 0

    for batch_idx, (data, targets) in enumerate(train_loader):
        data, targets = data.to(device), targets.to(device)

        # Forward
        outputs = model(data)
        loss = criterion(outputs, targets)

        # Backward
        optimizer.zero_grad()
        loss.backward()

        # Gradient clipping — prevents exploding gradients
        torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)

        optimizer.step()
        total_loss += loss.item()

    scheduler.step()
    avg_loss = total_loss / len(train_loader)
    print(f"Epoch {epoch+1}: loss={avg_loss:.4f}, lr={scheduler.get_last_lr()[0]:.6f}")

# Evaluation
model.eval()
with torch.no_grad():         # disable gradient computation
    for data, targets in test_loader:
        data, targets = data.to(device), targets.to(device)
        outputs = model(data)
        # compute accuracy, etc.
Two no_grad() variants: torch.no_grad() (explicit) disables all gradient tracking — use during evaluation. @torch.no_grad() decorator does the same. Both save memory and speed up inference.

6. Data Loading

from torch.utils.data import Dataset, DataLoader

class TextDataset(Dataset):
    def __init__(self, texts, labels, tokenizer, max_len=512):
        self.texts = texts
        self.labels = labels
        self.tokenizer = tokenizer
        self.max_len = max_len

    def __len__(self):
        return len(self.texts)

    def __getitem__(self, idx):
        encoding = self.tokenizer(
            self.texts[idx],
            truncation=True,
            max_length=self.max_len,
            padding='max_length',
            return_tensors='pt'
        )
        return {
            'input_ids': encoding['input_ids'].squeeze(0),
            'attention_mask': encoding['attention_mask'].squeeze(0),
            'label': torch.tensor(self.labels[idx], dtype=torch.long)
        }

# DataLoader wraps the dataset with batching, shuffling, padding
train_loader = DataLoader(
    TextDataset(texts, labels, tokenizer),
    batch_size=16,
    shuffle=True,
    num_workers=4,         # parallel data loading
    pin_memory=True,       # faster GPU transfer
    drop_last=True         # drop last incomplete batch
)

8. Fine-Tuning Strategies

Fine-tuning means taking a pretrained model and adapting it to your task. There are several strategies:

Strategy A — Full Fine-Tuning (All Parameters)

# Train all parameters — highest capacity, highest memory
for param in model.parameters():
    param.requires_grad = True

Strategy B — Feature Extraction (Freeze Backbone)

# Freeze all pretrained layers — only train the classifier head
for param in model.parameters():
    param.requires_grad = False

# Unfreeze and make classifier trainable
model.classifier.weight.requires_grad = True
model.classifier.bias.requires_grad = True

Strategy C — Differential Learning Rates

# Lower LR for pretrained layers (already close to optimal), higher for new head
optimizer = torch.optim.AdamW([
    {'params': model.pretrained_layers.parameters(), 'lr': 1e-5},
    {'params': model.new_head.parameters(),      'lr': 1e-3}
], weight_decay=0.01)

9. LoRA: Efficient Fine-Tuning

LoRA (Low-Rank Adaptation) is the most practical fine-tuning method for large models. Instead of updating all weights, it adds small trainable rank-decomposition matrices alongside frozen pretrained weights.

pip install peft bitsandbytes  # bitsandbytes for quantization
from peft import LoraConfig, get_peft_model, TaskType

# LoRA config
lora_config = LoraConfig(
    r=16,                                   # rank — higher = more capacity, more params
    lora_alpha=32,                          # scaling factor
    target_modules=["q_proj", "v_proj"],   # which layers to adapt (attention projections)
    lora_dropout=0.05,
    bias="none",
    task_type=TaskType.CAUSAL_LM
)

# Wrap the model with LoRA
model = get_peft_model(base_model, lora_config)
model.print_trainable_parameters()
# Output: "trainable params: 4M || all params: 7B || trainable%: 0.057%"
Why LoRA works: Pretrained models are already good at their domain — they just need task-specific direction. LoRA injects small, low-rank matrices that steer the model's behavior without overwriting its knowledge. A rank-8 or rank-16 LoRA can have under 1% of the parameters trainable.

LoRA Training Loop

# LoRA parameters are the only trainable ones — much smaller than full model
optimizer = torch.optim.AdamW(model.parameters(), lr=2e-4)

for epoch in range(num_epochs):
    model.train()
    for batch in train_loader:
        optimizer.zero_grad()
        outputs = model(**batch)
        loss = outputs.loss          # HuggingFace models return Loss under training
        loss.backward()
        optimizer.step()

# Save only the LoRA adapters (not the full model)
model.save_pretrained("./my-lora-adapter")

Merging LoRA Weights

# Merge LoRA weights into base model for inference (no LoRA overhead)
from peft import PeftModel
base_model = AutoModelForCausalLM.from_pretrained("base-model-name")

10. Practical Tips

  • Always set seeds for reproducibility: torch.manual_seed(42); torch.cuda.manual_seed_all(42)
  • Use torch.compile() (PyTorch 2.0+) to speed up training by 10-30%: model = torch.compile(model)
  • Gradient checkpointing trades compute for memory: model.gradient_checkpointing_enable()
  • Mixed precision training (torch.cuda.amp) cuts memory ~50% with minimal speed loss
  • DataLoader workers > 0 keeps GPU fed; pin_memory=True speeds host-to-GPU transfer
  • Use accelerate or DeepSpeed for multi-GPU / multi-node training at scale
  • Monitor GPU memory: torch.cuda.memory_allocated() / 1e9 to see what's using VRAM
  • Save best checkpoint: track validation loss, save model.state_dict() only when it's improved
  • For inference: always model.eval() + torch.no_grad() to disable dropout and gradient tracking
# Mixed precision training (PyTorch native)
from torch.cuda.amp import autocast, GradScaler

scaler = GradScaler()
for data, targets in train_loader:
    optimizer.zero_grad()
    with autocast():              # automatic FP16 computation
        outputs = model(data)
        loss = criterion(outputs, targets)
    scaler.scale(loss).backward()
    scaler.step(optimizer)
    scaler.update()
